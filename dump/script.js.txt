FUNC renderSnippet (snippet, delay) ->
	snippet << root div property
	snippet << 'visible' toggle property

	lazy render snippet after delay ms
		snippet.root.classSame << 'a-string'
		snippet.root.innerHTML << html string
		
		IF markdown snippet 
			class wrappable
			target _blank for inline links
			highlight text (?)
		FI

		set innerHTML <pre><code> with classes for editor & language

		highlight code 
 
		snippet.containerHighlight << highlighted code

		IF has fore/meta/aft links
			render and append to snippet.containerHighlight
		FI 

		append snippet.containerHighlight to snippet.root
		
		add language-specific danger zone listeners to container elements

		add generic listeners to root (copy, link, github, edit)

CNUF

	FI

	ELSE

		update URL params 
	FI
CNUF
